## Current Mappings (Ins Mode)

### right_command + key
- rcmd + j = word left
- rcmd + k = word right
- rcmd + m = line start
- rcmd + comma = line end
- rcmd + up = left arrow
- rcmd + down = right arrow
- rcmd alone = }
- rshift + rcmd alone = ]

### space + key
- space + j = shift+word left (select)
- space + k = shift+word right (select)
- space + m = shift+line start (select)
- space + comma = shift+line end (select)
- space + up = shift+left (select)
- space + down = shift+right (select)
- space + enter = cmd+enter

### quote (')
- quote hold = mirror mode

### page_up / page_down
- page_down = left click
- page_up = right click
